package kivipaperisakset;

/**
 *
 * @author Joona Nylander
 * 
 * Pelaajaa kuvaava luokka. Tallettaa pelaajalle kuuluvat tiedot omiin instanssimuuttujiin.
 */
public class Pelaaja {
	/**
	 * Voittojen lukumäärä
	 */
    int voitot;
    /**
     * Pelaajan valinta. Mahdolliset arvot ovat "kivi", "paperi" tai "sakset".
     */
    String valinta;

    /**
     * Valitse randomilla kivi, paperi tai sakset
     * @return Valinta string-muodossa.
     */
    public String pelaajanValinta() {
        valinta = "";
        int c = (int) (Math.random() * 3);
        switch (c) {
            case 0:
                valinta = ("kivi");
                break;
            case 1:
                valinta = ("paperi");
                break;
            case 2:
                valinta = ("sakset");
                break;
        }
        return valinta;
    }
    
    /**
     * Lisää pelaajan voittojen määrää yhdellä
     * @return Palauttaa voittojen määrän int-muodossa
     */
    public int setVoitot() {
        voitot++;
        return voitot;
    }
    
    /**
     * 
     * @return Palauttaa voitot
     */
    public int getVoitot() {
        return voitot;
    }
}
