package kivipaperisakset;

/**
 *
 * @author Joona Nylander
 * 
 * Pääohjelman luokka. Sisältää pelaa()-metodin ja main-metodin.
 */
public class Peli {
	
	/*
	 * Vertaa valintoja keskenään ja palauttaa voittavan valinnan.
	 */
	static String pelaa(String valinta1, String valinta2) {
		if (valinta1.equals("kivi")) {
			if (valinta2.equals("paperi")) {
				System.out.println("Pelaaja 2 voittaa");
				return valinta2;
			} else if (valinta2.equals("sakset")) {
				System.out.println("Pelaaja 1 voittaa");
				return valinta1;
			}
		} else if (valinta1.equals("paperi")) {
			if (valinta2.equals("sakset")) {
				System.out.println("Pelaaja 2 voittaa");
				return valinta2;
			} else if (valinta2.equals("kivi")) {
				System.out.println("Pelaaja 1 voittaa");
				return valinta1;
			}
		} else if (valinta1.equals("sakset")) {
			if (valinta2.equals("kivi")) {
				System.out.println("Pelaaja 2 voittaa");
				return valinta2;
			} else if (valinta2.equals("paperi")) {
				System.out.println("Pelaaja 1 voittaa");
				return valinta1;
			}
		}
		return "";
	}

	public static void main(String args[]) {
		Pelaaja p1 = new Pelaaja();
		Pelaaja p2 = new Pelaaja();
		boolean peliLoppui = false;
		int pelatutPelit = 0; // Pelattujen pelien lkm
		int p1Voitot = p1.voitot; // Pelaaja 1:n voittojen lkm
		int p2Voitot = p2.voitot; // Pelaaja 2:n voittojen lkm
		int tasapelit = 0; // Tasapelien lkm

		do {
			System.out.println("Erä: " + pelatutPelit + " =====================\n");
			System.out.println("Tasapelien lukumäärä: " + tasapelit + "\n");

			p1.pelaajanValinta();
			p2.pelaajanValinta();

			System.out.println("Pelaaja 1: " + p1.valinta + "\n\t Pelaaja 1:llä koossa " + p1Voitot + " voittoa.");
			System.out.println("Pelaaja 2: " + p2.valinta + "\n\t Pelaaja 2:lla koossa " + p2Voitot + " voittoa.");

			if (p1.valinta == p2.valinta) {
				tasapelit++;
				System.out.println("\n\t\t\t Tasapeli \n");
			} else {
				String lopputulos = pelaa(p1.valinta, p2.valinta);
				if (lopputulos.equals(p1.valinta)) {
					p1Voitot = p1.setVoitot();
				} else {
					p2Voitot = p2.setVoitot();
				}
			}

			pelatutPelit++;
			if ((p1.getVoitot() >= 3) || (p2.getVoitot() >= 3)) {
				peliLoppui = true;
				System.out.println("KOLME VOITTOA - PELI PÄÄTTYY");
				System.out.println("VOITTAJA: ");
			}
			System.out.println();
		} while (!peliLoppui);
	}
}
