package kivipaperisakset;

import static org.junit.jupiter.api.Assertions.*;

import java.util.Arrays;
import java.util.List;

import org.junit.jupiter.api.Test;

class Testi {
	
	/**
	 * Testaa luokan Pelaaja valinta()-metodin toimivuuden
	 */
	@Test
	void testValinta() {
		Pelaaja pelaaja = new Pelaaja();
		pelaaja.pelaajanValinta();
		String odotetutArvot[] = { "kivi", "sakset", "paperi" };
		List<String> odotetutArvotList = Arrays.asList(odotetutArvot);

		assertTrue(odotetutArvotList.contains(pelaaja.valinta));
	}
	
	
	/**
	 * Testaa luokan Pelaaja voittojen lisäämisen toimivuuden
	 */
	@Test
	void testVoitot() {
		Pelaaja pelaaja = new Pelaaja();
		assertEquals(0, pelaaja.voitot);
		pelaaja.setVoitot();
		pelaaja.setVoitot();
		assertEquals(2, pelaaja.voitot);
	}

	/**
	 * Testaa luokan Peli pelaa()-metodin toimivuuden
	 */
	@Test
	void testPelaaPaperi() {
		String valinta1 = "paperi";
		String valinta2 = "sakset";

		String lopputulos = Peli.pelaa(valinta1, valinta2);
		assertEquals(valinta2, lopputulos);
		
		valinta2 = "kivi";
		lopputulos = Peli.pelaa(valinta1, valinta2);
		assertEquals(valinta1, lopputulos);
	}
	
	/**
	 * Testaa luokan Peli pelaa()-metodin toimivuuden
	 */
	@Test

		String valinta2 = "sakset";

		valinta2 = "paperi";
		lopputulos = Peli.pelaa(valinta1, valinta2);
		assertEquals(valinta2, lopputulos);
	}
	
	/**
	 * Testaa luokan Peli pelaa()-metodin toimivuuden
	 */
	@Test

		String valinta1 = "sakset";
		String valinta2 = "paperi";

	 * Testaa luokan Peli pelaa()-metodin toimivuuden

		String valinta2 = "kivi";

		assertEquals("", lopputulos);
	}

}
